org 26c
arr: WORD 5E8
outsymb: WORD 0x0A
start: CLA

s1: IN 0x19
	AND	#0x40	
	BEQ	s1
	IN	0x18
	CMP outsymb
	BEQ STOP1
	ST (arr)

s2: IN 0x19
	AND	#0x40	
	BEQ	s2
	IN	0x18
	CMP outsymb
	BEQ STOP2
	SWAB
	ADD (arr)
	ST (arr)
	
	LD arr
	INC
	ST arr
	
	JUMP s1
	
STOP1:
	ST (arr)
	JUMP out1

STOP2:
	LD arr
	INC
	ST arr
	ST (arr)
	JUMP out1

out1:
	LD (arr)
	SWAB
	OUT 0xC
	SWAB
	OUT 0xC
	CLA
	LD arr
	CMP outsymb
	BEQ EXIT
	LD -(arr) 
	JUMP out1

 EXIT: HLT
